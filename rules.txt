
/header/  

<!DOCTYPE html>
<html>
<head>
<title>Page Title</title>
</head>
<body>

/header/

begin{

	ITAlIC -> * text *  -> <em> text </em>
	BOLD -> ** text ** -> <strong> text </strong>
	UNDERLINE -> % text % -> <u> text </u>
	PARAGRAPH -> & text -> <p> text </p>
	TITLE1 -> # text -> <h1> text </h1>
	TITLE2 -> ## text -> <h2> text </h2>
	TITLE3 -> ### text -> <h3> text </h3>
	TITLE4 -> #### text -> <h4> text </h4>
	
	nested{
		LIST ->   text  ->  <ul> text </ul>
		ITEM -> + text -> <li> text </li>
	}
	
	nested{
		ORDERLIST -> text  ->  <ol> text </ol>
		ELEMENTO -> $ text -> <li> text </li>
		
	}
	
	LABEL -> @ text @ -> <a_name=" text ">
	
	LINK -> [ text ] ( literal ) -> <a_href=" literal "> text </a>
	IMG -> ![ text ] ( literal ) -> <img_src=" literal "_alt=" text "></img>
	
}

/footer/
</body>
</html>
/footer/
